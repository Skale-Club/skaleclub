    **Helper — keyword overlap (file-local):**
    ```typescript
    function scoreKeywordOverlap(item: BlogRssItem, keywords: string[]): number {
      if (keywords.length === 0) return 0; // empty seoKeywords -&gt; no signal, NOT 1
      const haystack = `${item.title} ${item.summary ?? ""}`.toLowerCase();
      const matches = keywords.filter((k) =&gt; haystack.includes(k)).length;
      return Math.min(matches / keywords.length, 1);
    }
    ```
    NOTE D-08 says "matches in title+summary / seoKeywords.length" — interpreted as the count of DISTINCT keywords that appear (not total occurrences of each keyword). Use `.filter((k) =&gt; haystack.includes(k)).length`, NOT a regex match-count over each keyword.

    **Helper — recency (file-local):**
    ```typescript
    function scoreRecency(item: BlogRssItem, now: Date): number {
      if (!item.publishedAt) return 0; // no date -&gt; treat as old
      const ageMs = now.getTime() - item.publishedAt.getTime();
      const ratio = Math.max(0, Math.min(ageMs / RECENCY_WINDOW_MS, 1));
      return 1 - ratio;
    }
    ```
    D-08 formula: `1 - clamp((now - published_at) / 14days, 0, 1)`. Today -&gt; ratio≈0 -&gt; recency≈1. 14d+ old -&gt; ratio=1 -&gt; recency=0. Future-dated (rare): negative age clamped to 0 -&gt; recency=1.

    **Public API — scoreItem (D-08):**
    ```typescript
    export function scoreItem(
      item: BlogRssItem,
      settings: BlogSettings,
      now: Date = new Date(),
    ): number {
      const keywords = parseSeoKeywords(settings.seoKeywords);
      const keywordScore = scoreKeywordOverlap(item, keywords);
      const recencyScore = scoreRecency(item, now);
      return KEYWORD_WEIGHT * keywordScore + RECENCY_WEIGHT * recencyScore;
    }
    ```
    `now` parameter is exposed for testability (the generator's deps already inject `now: () =&gt; new Date()` — Plan 03 should pass that through, but the default keeps the signature ergonomic).

    **Public API — selectNextRssItem (D-09):**
    ```typescript
    export async function selectNextRssItem(
      settings: BlogSettings,
      now: Date = new Date(),
    ): Promise&lt;BlogRssItem | null&gt; {
      const candidates = await storage.listPendingRssItems(PENDING_BATCH_SIZE);
      if (candidates.length === 0) {
        console.log("[rss-selector] no pending items — generator should skip");
        return null;
      }

      let best: BlogRssItem | null = null;
      let bestScore = -Infinity;

      // candidates already ordered by publishedAt DESC NULLS LAST.
      // We iterate in that order and use STRICT &gt; on score, so on ties
      // the FIRST candidate (newer publishedAt) wins — implicit tiebreak.
      for (const candidate of candidates) {
        const score = scoreItem(candidate, settings, now);
        if (score &gt; bestScore) {
          bestScore = score;
          best = candidate;
        }
      }

      if (best) {
        console.log(
          `[rss-selector] picked item id=${best.id} score=${bestScore.toFixed(3)} title=${best.title.slice(0, 80)}`,
        );
      }

      return best;
    }
    ```
