Create `server/routes/notifications.ts` following the skaleHub.ts pattern exactly:

```typescript
import type { Express } from "express";
import { z } from "zod";
import { storage } from "../storage.js";
import { insertNotificationTemplateSchema } from "#shared/schema.js";
import { requireAdmin } from "./_shared.js";

export function registerNotificationRoutes(app: Express) {
  app.get("/api/notifications/templates", requireAdmin, async (_req, res) => {
    try {
      return res.json(await storage.getNotificationTemplates());
    } catch (err) {
      return res.status(500).json({ message: (err as Error).message });
    }
  });

  app.put("/api/notifications/templates/:id", requireAdmin, async (req, res) => {
    try {
      const id = Number(req.params.id);
      if (!Number.isInteger(id) || id <= 0) {
        return res.status(400).json({ message: "Invalid template id" });
      }
      const updateSchema = insertNotificationTemplateSchema.partial();
      const parsed = updateSchema.safeParse(req.body);
      if (!parsed.success) {
        return res.status(400).json({ message: parsed.error.errors[0].message });
      }
      const updated = await storage.upsertNotificationTemplate({ ...parsed.data, id });
      return res.json(updated);
    } catch (err) {
      return res.status(500).json({ message: (err as Error).message });
    }
  });
}
```

Key requirements:
- `import { z } from "zod"` is required because `.partial()` is called on the schema inline (Pitfall 6 — z must be imported even though skaleHub.ts doesn't need it)
- Use `insertNotificationTemplateSchema.partial()` (NOT the full schema) for PUT — only `body` and `active` need to be accepted; `channel` and `eventKey` are already set in the DB row and must not be overwritten
- `id` comes from `req.params.id` (parsed to Number), then merged as `{ ...parsed.data, id }` into the storage call
- The storage `upsertNotificationTemplate` uses an id-present check internally (update path when id exists)
- No `.strict()` on the schema — unknown extra keys are silently ignored
- `requireAdmin` is imported from `./_shared.js` (NOT from the inline function in routes.ts)
