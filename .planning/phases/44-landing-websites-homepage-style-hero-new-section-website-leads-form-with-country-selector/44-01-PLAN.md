```ts
import { z } from "zod";

export const heroWebsitesPropsSchema = z.object({
  headline:            z.string().optional(),
  subheadline:         z.string().optional(),
  ctaLabel:            z.string().optional(),
  backgroundImageUrl:  z.string().url().optional(),
});
export type HeroWebsitesProps = z.infer<typeof heroWebsitesPropsSchema>;

const DEFAULTS = {
  headline:    "Sites que vendem para o seu negócio",
  subheadline: "Da página de captura ao e-commerce: criamos sites rápidos, otimizados para Google e prontos para converter visitantes em clientes.",
  ctaLabel:    "Quero meu site",
} as const;

export function HeroWebsitesSection({ props }: { props: HeroWebsitesProps }) {
  const headline    = props.headline    ?? DEFAULTS.headline;
  const subheadline = props.subheadline ?? DEFAULTS.subheadline;
  const ctaLabel    = props.ctaLabel    ?? DEFAULTS.ctaLabel;
  const bgUrl       = props.backgroundImageUrl;

  const handleCtaClick = () => {
    const trigger = document.querySelector<HTMLElement>('[data-landing-lead-cta]');
    if (trigger) {
      trigger.click();
    } else {
      // Fallback: scroll to bottom (where leadFormCta typically lives)
      window.scrollTo({ top: document.body.scrollHeight, behavior: "smooth" });
    }
  };

  return (
    <section
      className="relative flex items-end pt-28 sm:pt-24 lg:pt-16 pb-36 sm:pb-48 lg:pb-4 overflow-hidden bg-[#1C53A3] min-h-[70vh] sm:min-h-[55vh] lg:min-h-[550px]"
      data-testid="section-hero-websites"
    >
      <div className="container-custom mx-auto relative z-10 w-full">
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-1 sm:gap-6 lg:gap-8 items-end">
          <div className="order-1 lg:order-2 text-white pt-6 sm:pt-8 lg:pt-16 pb-16 sm:pb-24 lg:pb-32 relative z-20">
            <h1 className="text-[9vw] sm:text-5xl md:text-6xl lg:text-4xl xl:text-5xl font-bold mb-3 lg:mb-6 font-display leading-[1.05] sm:leading-[1.1]">
              <span className="text-transparent bg-clip-text bg-gradient-to-r from-white to-blue-200">
                {headline}
              </span>
            </h1>
            <p className="text-base sm:text-xl text-blue-50/80 mb-4 lg:mb-8 leading-relaxed max-w-xl">
              {subheadline}
            </p>
            <div className="flex flex-col sm:flex-row gap-3 lg:gap-5 flex-wrap">
              <button
                type="button"
                onClick={handleCtaClick}
                data-testid="button-hero-websites-cta"
                className="w-full sm:w-auto shrink-0 px-6 sm:px-8 py-3 sm:py-4 bg-[#FFFF01] hover:bg-[#e6e600] hover:scale-105 text-black font-bold rounded-full transition-all flex items-center justify-center gap-2 text-base sm:text-lg whitespace-nowrap"
              >
                {ctaLabel}
              </button>
            </div>
          </div>
          <div className="order-2 lg:order-1 relative flex h-full items-end justify-center lg:justify-end self-end w-full lg:min-h-[400px] z-10 lg:ml-[-3%]">
            {bgUrl ? (
              <img
                src={bgUrl}
                alt=""
                className="w-[92vw] sm:w-[98%] lg:w-full max-w-[380px] sm:max-w-[360px] md:max-w-[430px] lg:max-w-[500px] xl:max-w-[560px] object-contain drop-shadow-2xl origin-bottom"
              />
            ) : null}
          </div>
        </div>
      </div>

      <div
        className="absolute inset-0"
        style={{
          background: `
            radial-gradient(ellipse at center, transparent 40%, rgba(0, 0, 0, 0.35) 80%, rgba(0, 0, 0, 0.85) 100%),
            radial-gradient(circle at 65% 10%, rgba(100, 135, 215, 0.30) 0%, transparent 60%),
            linear-gradient(
              to right bottom,
              #09152d, #0b152a, #0d1427, #0f1424, #101421, #121622,
              #151723, #171924, #1c1c29, #21202e, #262332, #2c2637
            )
          `,
        }}
      />
    </section>
  );
}
```

**Rationale on the design choices:**

- **Brand-blue background + gradient overlay** mirrors HeroSection.tsx:16 and the gradient at HeroSection.tsx:67-90. This is the homepage tone.
- **Yellow pill CTA** (`bg-[#FFFF01] text-black rounded-full`) honors the CLAUDE.md Brand Guidelines explicitly (Yellow with black bold text, pill-shaped). The Home hero currently uses the blue accent `#406EF1` button — we intentionally diverge here because Brand Guidelines specify yellow for CTAs and this is a fresh page with no migration cost.
- **Headline gradient text** (`text-transparent bg-clip-text bg-gradient-to-r from-white to-blue-200`) copies HeroSection.tsx:31 so the typography feels cohesive.
- **No `useTranslation` / no `t()` calls** — the copy is authored in pt-BR directly per CONTEXT.md "Form question copy and landing section copy authored in pt-BR." There are NO new string literals that need PT translations added to `translations.ts` (per CLAUDE.md Translation Rules — those rules apply to `t()` literals only).
- **No `data-form-trigger="lead-form"` attribute** — that's the Home's mechanism (handled in `App.tsx`'s global click listener). On dynamic landings the bridge is `data-landing-lead-cta` instead, mirroring `HeroSectionAdapter.tsx:23-29`.
- **File size target:** ~110 lines. Hard cap from CLAUDE.md: 600.
