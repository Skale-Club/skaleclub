**2. Create scripts/seed-skale-hub-group-landing.ts** — idempotent upsert:

```ts
// Seed the existing Skale Hub WhatsApp group landing into the managed
// landing_pages table. Idempotent: re-running updates the row in place.
import { pool } from "../server/db.js";
import { db } from "../server/db.js";
import { landingPages } from "../shared/schema/landings.js";
import { eq } from "drizzle-orm";

const SLUG = "grupo";

async function seed() {
  console.log(`Seeding managed landing: slug='${SLUG}'`);
  const sections = [{ type: "whatsappGroup", props: {} }];
  const existing = await db.select().from(landingPages).where(eq(landingPages.slug, SLUG));

  if (existing.length > 0) {
    await db
      .update(landingPages)
      .set({
        name: "Skale Hub WhatsApp Group",
        sections,
        isActive: true,
        updatedAt: new Date(),
      })
      .where(eq(landingPages.slug, SLUG));
    console.log(`Updated existing landing (id=${existing[0].id}).`);
  } else {
    const [row] = await db
      .insert(landingPages)
      .values({
        slug: SLUG,
        name: "Skale Hub WhatsApp Group",
        sections,
        isActive: true,
      })
      .returning();
    console.log(`Inserted new landing (id=${row.id}).`);
  }

  await pool.end();
}

seed().catch((err) => {
  console.error("Seed failed:", err);
  process.exit(1);
});
```

Run it: `npx tsx scripts/seed-skale-hub-group-landing.ts` — expect "Inserted new landing (id=…)" on first run, "Updated existing landing (id=…)" on subsequent runs.
