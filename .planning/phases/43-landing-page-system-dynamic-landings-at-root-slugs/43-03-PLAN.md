**Pattern:** One file per section type, each file is a thin adapter that:
1. Declares a `propsSchema` (zod) for the JSONB `props` object.
2. Exports a component that takes the parsed `props` and renders the existing home/* primitive.

Do NOT modify the home/* primitives themselves. The adapter's job is to translate the JSONB shape to the existing component's prop signature.

For sections that need live data from `/api/company-settings` (Hero, Services, Reviews, Blog, About, AreasServed), the adapter is allowed to `useQuery` for `CompanySettings` and pass the slices its underlying component needs. Mirror the way Home.tsx already does this — search `client/src/pages/Home.tsx` for `useQuery<CompanySettings>` to confirm the pattern.

Build the public renderer. The route param `slug` is supplied by Wouter via `useParams`.

```tsx
import { useParams } from "wouter";
import { useQuery } from "@tanstack/react-query";
import { lazy, Suspense } from "react";
import { PageLoader } from "@/components/ui/spinner";
import { sectionRegistry } from "@/components/landings/sectionRegistry";

const NotFound = lazy(() => import("@/pages/not-found"));

interface LandingResponse {
  slug: string;
  name: string;
  sections: Array<{ type: string; props: Record<string, unknown> }>;
  isActive: boolean;
}

export default function DynamicLanding() {
  const { slug } = useParams<{ slug: string }>();

  const { data, isLoading, error } = useQuery<LandingResponse>({
    queryKey: [`/api/landing-pages/slug/${slug}`],
    enabled: !!slug,
    retry: false,
  });

  if (isLoading) return <PageLoader />;

  if (error || !data) {
    return (
      <Suspense fallback={<PageLoader />}>
        <NotFound />
      </Suspense>
    );
  }

  return (
    <>
      {data.sections.map((section, idx) => {
        const entry = sectionRegistry[section.type];
        if (!entry) {
          // Unknown section type — skip silently in prod, surface inline in dev
          if (import.meta.env.DEV) {
            return (
              <div key={idx} className="border border-red-500/40 bg-red-500/5 p-4 text-sm text-red-300">
                Unknown section type: <code>{section.type}</code>
              </div>
            );
          }
          return null;
        }
        const parsed = entry.propsSchema.safeParse(section.props ?? {});
        if (!parsed.success) {
          if (import.meta.env.DEV) {
            return (
              <div key={idx} className="border border-amber-500/40 bg-amber-500/5 p-4 text-sm text-amber-300">
                Invalid props for section <code>{section.type}</code> (index {idx}): {parsed.error.message}
              </div>
            );
          }
          return null;
        }
        const Component = entry.component;
        return <Component key={idx} props={parsed.data} />;
      })}
    </>
  );
}
```
