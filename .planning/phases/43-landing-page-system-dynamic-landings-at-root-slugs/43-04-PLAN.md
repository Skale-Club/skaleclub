**LandingsSection.tsx** (≤ 200 lines) — top-level wrapper. Manages two views: list + editor. Uses `useState<string | null>` to track the editing landing id; when null, shows the list; when set, dynamically renders `<LandingEditor landingId={…} onBack={() => setEditingId(null)} />` (LandingEditor is created in Task 3).

Skeleton:
```tsx
import { useState } from "react";
import { Layers } from "lucide-react";
import { SectionHeader } from "@/components/admin/shared";
import { LandingsList } from "./LandingsList";
import { LandingEditor } from "./LandingEditor";
import { CreateLandingDialog } from "./CreateLandingDialog";

export function LandingsSection() {
  const [editingId, setEditingId] = useState<string | null>(null);
  const [createOpen, setCreateOpen] = useState(false);

  if (editingId) {
    return <LandingEditor landingId={editingId} onBack={() => setEditingId(null)} />;
  }
  return (
    <>
      <SectionHeader
        title="Landings"
        description="Managed landing pages — addressable at any /slug, composed from a section registry."
        icon={<Layers className="w-5 h-5" />}
        action={<button data-testid="button-create-landing" onClick={() => setCreateOpen(true)} className="...">New landing</button>}
      />
      <LandingsList onEdit={setEditingId} />
      <CreateLandingDialog
        open={createOpen}
        onOpenChange={setCreateOpen}
        onCreated={(id) => { setCreateOpen(false); setEditingId(id); }}
      />
    </>
  );
}
```

Use the same button styling found in FormsSection.tsx for "New form" so the look is consistent.
